CREATE DATABASE script CHARACTER SET utf8 COLLATE utf8_general_ci;
CREATE TABLE member (
    id int AUTO_INCREMENT,
    usr VARCHAR(40) not null,
    pwd VARCHAR(20) not null,
    mob VARCHAR(10) not null,
    email VARCHAR(100) not null,
    primary key(id)
);
